SELECT FILMCOD,FILMTIT,REANOM,REAPRE FROM FILM NATURAL JOIN REALISE WHERE REANOM LIKE CONCAT('&debutNom', '%') ORDER BY REACOD ASC;

--15:
SELECT FILMCOD,FILMTIT,FILMAN FROM FILM NATURAL JOIN JOUER WHERE JOUER.ACTCOD = &codeActeur AND FILM.REACOD = &codeReal;

--17:
SELECT FILM.FILMCOD,FILM.FILMTIT FROM FILM INNER JOIN JOUER J1 ON J1.FILMCOD = FILM.FILMCOD INNER JOIN JOUER J2 ON J2.FILMCOD = FILM.FILMCOD WHERE J1.ACTCOD = &codeActeur1 AND J2.ACTCOD = &codeActeur2 AND J1.FILMCOD = J2.FILMCOD;

--19:
SELECT FILM.FILMTIT,FILM.FILMAN,FILM.FILNAT,CATEGORIE.CATLIB,REALISE.REANOM,REALISE.REAPRE,ACTEUR.ACTNOM,ACTEUR.ACTPRE,K7.K7COD,K7.K7TYP,K7.K7LOC
FROM FILM
    LEFT OUTER JOIN CATEGORIE ON FILM.CATCOD = CATEGORIE.CATCOD
    INNER JOIN REALISE ON FILM.REACOD = REALISE.REACOD
    INNER JOIN JOUER ON JOUER.FILMCOD = FILM.FILMCOD
    INNER JOIN ACTEUR ON ACTEUR.ACTCOD = JOUER.ACTCOD
    INNER JOIN K7 ON K7.FILMCOD = FILM.FILMCOD
WHERE
    FILM.FILMCOD = &codeFilm;
